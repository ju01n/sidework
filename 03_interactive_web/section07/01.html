// 생성자 (constructor) function Person(nickname, age){ this.nickname = nickname; this.age = age; this.introduce = function(){ console.log('안녕하세요? 저는' + this.nickname + '이고, 나이는' + this.age + '살이에여') } } // 인스턴스 (instance) const person1 = new Person('일분이', 10); // new : 개별 객체를 this가 가리키게됨 const person2 = new Person('이분이', 8); person1.introduce(); person2.introduce();
